
\subsubsection{\texttt{get\_source\_file}}

returns the source file path of the declaration of a base-level program feature reflected by a \meta{Declaration}.

\begin{minted}[tabsize=4]{cpp}

template <typename T>
requires Declaration<T>
struct get_source_file : StringConstant { };
template <typename T>
using get_source_file_t = typename get_source_file<T>::type;

\end{minted}
