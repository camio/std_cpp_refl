
\subsubsection{\texttt{get\_scope}}

returns the Scope reflecting the scope of a scoped declaration reflected by a \meta{Scoped}.

\begin{minted}{cpp}
namespace meta {
\end{minted}
\begin{minted}[xleftmargin=1em,tabsize=4]{cpp}
template <Scoped T>
struct get_scope
{
	typedef /* generated by the compiler */ type;
	static_assert(Scope<type>, "");
};
	
template <Scoped T>
constexpr auto get_scope_v = get_scope<T>::value;
\end{minted}
\begin{minted}{cpp}
} // namespace meta
\end{minted}
