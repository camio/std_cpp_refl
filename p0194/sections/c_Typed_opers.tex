
\subsubsection{\texttt{get\_type}}

returns the Type reflecting the type of base-level declaration with a type reflected by a \meta{Typed}.

\begin{minted}[tabsize=4]{cpp}

template <typename T>
requires Typed<T>
struct get_type
{
	typedef Type type;
};
	
template <typename T>
using get_type_t = typename get_type<T>::type;

\end{minted}
