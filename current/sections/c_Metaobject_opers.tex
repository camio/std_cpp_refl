
\subsubsection{\texttt{source\_file}}

returns a source file path of the declaration of a base-level program feature reflected by a \meta{object}.

\begin{minted}[tabsize=4]{cpp}

template <typename T>
requires Metaobject<T>
struct source_file
{
	typedef const char value_type[N+1];
	static constexpr const char value[N+1];
	typedef StringConstant type;

	operator const char* (void) const noexcept;
	const char* operator (void) const noexcept;
};


template <typename T>
using source_file_t = typename source_file<T>::type;

\end{minted}
